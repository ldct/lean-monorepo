\section*{Bernoulli Trial 2025 (questions 10-16)}

\begin{enumerate}

\item (4 minutes) For any positive integer $n$, let $S_n$ denote the symmetric group of all permutations of $\{1,\ldots,n\}$.  
For each $\sigma \in S_n$, let $\mathrm{Orb}(\sigma)$ be the number of cycles of $\sigma$.

\textbf{T/F:}
\[
\frac{1}{69!} \sum_{\sigma \in S_{69}} \mathrm{Orb}(\sigma) < 4.
\]

\item (3 minutes) \textbf{T/F:} There does not exist $B \in M_{69\times 69}(\mathbb{R})$ such that
\[
\dim_\mathbb{R}\big( \{ BAB : A \in M_{69\times 69}(\mathbb{R}) \} \big)
= 2025.
\]

\item (3 minutes) \textbf{T/F:} There exist $a,b,c \in \mathbb{Z}$ such that
\[
\left| \zeta_{13}^a + \zeta_{13}^b + \zeta_{13}^c + 1 \right|
= \sqrt{3},
\qquad \zeta_{13} = e^{2\pi i/13}.
\]

\item (4 minutes) \textbf{T/F:}
\[
\int_0^1 \left( 
\sqrt{1 + 8x - 8x^3} - 
\frac{\sqrt{x}}{4 - x + 1} -
\frac{1}{4x}
\right)\, dx \notin \mathbb{Q}.
\]

\item (4 minutes) A positive integer is a \emph{Gian's integer} if it is of the form $a^4 + b^3$ for some positive integers $a,b$.

\textbf{T/F:} For any integer $n \ge 3$, there exist infinitely many integers $m$ such that there are exactly $n+1$ Gian's integers among
\[
m+1,\; m+2,\; \ldots,\; m+n^3.
\]

\item (5 minutes) For any positive integer $m$, let $S(m)$ be the number of positive integers $n < \mathrm{lcm}(1,2,\ldots,m)$ such that the remainders of $n$ modulo $2,3,\ldots,m$ are all distinct.

\textbf{T/F:} $S(2025)-1$ is a power of $2$.

\item (Tie-break, 3 minutes) Compute
\[
\sum_{k=1}^{8} e^{-k \cdot 2\pi/9}.
\]

\end{enumerate}
