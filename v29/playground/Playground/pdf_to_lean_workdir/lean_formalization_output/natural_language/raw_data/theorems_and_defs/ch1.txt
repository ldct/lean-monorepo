\section{Definitions}

\begin{definition}[Definition 1.1, Concrete group]
A set $G$ is a group iff it is the set of symmetries of something.
\end{definition}

\begin{definition}[Definition 1.2, Abstract group]
A set $G$ with a binary relation (if $a, b \in G$, written $a \times b$, $a + b$, $a \cdot b$, $a \circ b$ or just $ab$) is a group iff this relation is associative, each element has an inverse, and there is an identity. In other words, for all $a, b, c \in G$,
1. There exists $e \in G$ such that $ea = ae = a$
2. There exists $a^{-1} \in G$ such that $aa^{-1} = a^{-1}a = e$
3. $a(bc) = (ab)c$
\end{definition}

\begin{definition}[Definition 1.3, Subgroup]
Let $G$ be a group. A subset $H \subseteq G$ is a subgroup of $G$ iff it contains the identity of $G$, is closed under the law of composition, and contains all inverses.
\end{definition}

\begin{definition}[Definition 1.4, Homomorphism]
Let $G, H$ be two groups. Then a function $f : G \to H$ is a homomorphism iff for all $a, b \in G$,
\[ f(ab) = f(a)f(b) \]
\end{definition}

\begin{theorem}[Proposition 1.1]
Homomorphisms preserve inverses and the identity.
\end{theorem}

\begin{definition}[Definition 1.5, Isomorphism]
A homomorphism is an isomorphism iff it is a bijection.
\end{definition}

\begin{definition}[Definition 1.6, Endomorphism]
A homomorphism is an endomorphism iff it has the same domain as its codomain.
\end{definition}

\begin{definition}[Definition 1.7, Automorphism]
A homomorphism is an automorphism iff it is an isomorphism and an endomorphism.
\end{definition}

\begin{definition}[Definition 1.8, Left action]
Let $G$ be a group and $S$ a set. Then a map $\cdot : G \times S \to S$ is a left action of $G$ on $S$ iff for all $s \in S$ and all $g, h \in G$ we have:
1. $g \cdot (h \cdot s) = (gh) \cdot s$
2. $e \cdot s = s$
where $e$ is the identity for $G$. A right action is a map $\cdot : S \times G \to S$ which satisfies the analogous properties.
\end{definition}

\begin{definition}[Definition 1.9, G-set]
Let $G$ be a group. A set $S$ is a $G$-set iff we have an action of $G$ on $S$ which is a homomorphism
\[ \pi : G \to \mathrm{Perm}(S) \]
\end{definition}

\begin{theorem}[Theorem 1.1]
A set $G$ is an abstract group iff it is a concrete group.
\end{theorem}

\begin{lemma}[Lemma 1.1]
If we have a left action of a group $G$ on a set $S$, then we automatically have a right action and vice versa. In particular, we write
\[ s \cdot g = g^{-1}s \]
to define a right action in terms of a given left action.
\end{lemma}

\begin{theorem}[Proposition 1.2]
There are 8 types of actions of a group on itself. There are four right, and four left actions on a group. The left four are:
1. Trivial: $g \cdot s = s$
2. Left: $g \cdot s = gs$
3. Right (to Left): $g \cdot s = sg^{-1}$
4. Adjoint (Conjugation): $g \cdot s = gsg^{-1}$
\end{theorem}

\section{Lagrange's theorem and product groups}

\begin{theorem}[Proposition 1.3]
The only group of order 1 is the trivial group.
\end{theorem}

\begin{definition}[Definition 1.10, Coset]
Let $G$ be a group and $H \subseteq G$ a subgroup. Then the left cosets of $H$ are sets of the form $gH = \{gh \mid h \in H\}$ for any $g \in G$. The right cosets of $H$ are sets of the form $Hg = \{hg \mid h \in H\}$ for any $g \in G$.
\end{definition}

\begin{theorem}[Theorem 1.2, Lagrange's theorem]
Let $G$ be a group, and $H \subseteq G$ be a subgroup. If both $|H|, |G|$ are finite, then $|H|$ divides $|G|$.
\end{theorem}

\begin{corollary}[Corollary 1.1]
There is only one group of prime order.
\end{corollary}

\begin{theorem}[Theorem 1.3, Fermat's little theorem]
Let $a, p \in \mathbb{Z}$ and $p$ prime, then
\[ a^{p-1} \equiv 1 \pmod{p} \]
\end{theorem}

\begin{definition}[Definition 1.11, Euler totient function]
A function $\varphi : \mathbb{Z} \to \mathbb{N}$ is called the Euler totient function iff it returns the number of integers between the argument and zero, which are relatively prime to the argument.
\end{definition}

\begin{theorem}[Proposition 1.4]
If $\varphi$ is Euler's totient function and $n \in \mathbb{Z}$, then
\[ \varphi(n) = n \prod_{p \mid n} \left(1 - \frac{1}{p}\right) \]
where the product is over all distinct prime integers $p$ dividing $n$.
\end{theorem}

\begin{theorem}[Theorem 1.4, Euler's little theorem]
Take integers $g, m \in \mathbb{Z}$ such that $(g, m) = 1$. It is then the case that
\[ a^{\varphi(n)} \equiv 1 \pmod{n} \]
where $\varphi$ is the Euler's totient function.
\end{theorem}

\begin{lemma}[Lemma 1.2]
If all elements have order 2 the group is abelian.
\end{lemma}

\begin{theorem}[Proposition 1.5]
There are only two groups of order 4. In particular, they are $\mathbb{Z}/4\mathbb{Z}$ and $\mathbb{Z}/2\mathbb{Z} \times \mathbb{Z}/2\mathbb{Z}$.
\end{theorem}

\begin{theorem}[Proposition 1.6]
The group of order 5 is unique up to isomorphism.
\end{theorem}

\begin{definition}[Definition 1.12, Product]
Let $G, H$ be groups. The group $G \times H$ is the product of $G$ and $H$ iff it is the cartesian product of $G$ and $H$ equipped with a binary relation given by the component operations of $G$ and $H$. Explicitly:
\[ (g_1, h_1)(g_2, h_2) = (g_1g_2, h_1h_2) \]
for all $(g_1, h_1), (g_2, h_2) \in G \times H$.
\end{definition}

\begin{theorem}[Proposition 1.7]
To see if $G = H_1 \times H_2$ we have to check:
1. Anything in $G$ is of the form $h_1 \times h_2$ for $h_1 \in H_1, h_2 \in H_2$.
2. $H_1 \cap H_2 = e$.
3. $H_1$ commutes with $H_2$.
\end{theorem}

\section{Quotient groups}

\begin{definition}[Definition 1.13, Normal subgroup]
Let $G$ be a group, and $H$ a subgroup. Then $H$ is normal iff for all $g \in G$, $gH = Hg$.
\end{definition}

\begin{theorem}[Proposition 1.8]
The cosets of a normal subgroup form a group under the multiplication of representative elements.
\end{theorem}

\begin{definition}[Definition 1.14, Quotient group]
Let $N \subseteq G$ be a normal subgroup. Then $G/N$ is the quotient group of $G$ by $N$.
\end{definition}

\begin{theorem}[Theorem 1.5]
Any subgroup of index 2 is normal.
\end{theorem}

\begin{theorem}[Proposition 1.9]
There is 1 abelian, cyclic group of order 6. It also splits as the following product:
\[ \mathbb{Z}/6\mathbb{Z} \cong \mathbb{Z}/2\mathbb{Z} \times \mathbb{Z}/3\mathbb{Z} \]
\end{theorem}

\begin{theorem}[Proposition 1.10]
Let $G = \mathbb{Z}/6\mathbb{Z}$ and $A = \{0, 2, 4\}$, $B = \{0, 3\}$. We have that $G \cong A \times B$.
\end{theorem}

\begin{theorem}[Proposition 1.11]
The other group of order 6 is the symmetric group $S_3$.
\end{theorem}

\begin{definition}[Definition 1.15, Center]
Let $G$ be a group. Then a subgroup $Z \subseteq G$ is the center of $G$ iff it consists of all the elements of $G$ which commute with all other elements of $G$.
\end{definition}

\begin{definition}[Definition 1.16, Orbit and Stabilizer]
Let $G$ be a group, and $S$ be a $G$-set. Then the orbit of an element $s \in S$ is
\[ Gs = \{g \cdot s \mid g \in G\} \]
The stabilizer of $s \in S$ is
\[ G_s = \{g \in G \mid g \cdot s = s\} \]
\end{definition}

\begin{theorem}[Proposition 1.12]
The orbits of a group $G$ on a $G$-set $S$ form a partition of $S$.
\end{theorem}

\begin{theorem}[Theorem 1.6, Cauchy's theorem]
Suppose $p$ is prime, and $p$ divides the order of a group $G$. Then there exists $a \in G$ such that the order of $a$ is $p$.
\end{theorem}

\begin{definition}[Definition 1.17, Direct product]
Let $A, B$ be two normal subgroups of a group $G$. The direct product of $A, B$ is the cartesian product with componentwise multiplication:
\[ (a_1, b_1)(a_2, b_2) := (a_1a_2, b_1b_2) \]
as inherited from $G$.
\end{definition}

\begin{theorem}[Proposition 1.13]
The direct product commutes.
\end{theorem}

\begin{definition}[Definition 1.18, Semidirect product]
Let $A$ be a normal subgroup of a group $G$. Let $B$ be another subgroup of $G$, which is not necessarily normal. For each $b \in B$ the map $a \mapsto bab^{-1}$ is an automorphism of $A$. Then if we write each such automorphism as $\varphi_b$ for $b \in B$, then take $\varphi_{b_1}\varphi_{b_2} := \varphi_{b_1b_2}$ for $b_1, b_2 \in B$, so we have a homomorphism from $B$ to $\mathrm{Aut}(A)$. The semidirect product of $A$ and $B$, written $A \rtimes B$, is then the cartesian product of the sets $A, B$ equipped with the operation:
\[ (a, b)(a', b') := (a \varphi_b(a'), bb') \]
\end{definition}

\begin{definition}[Definition 1.19, Exact sequence]
Let $A, B, C$ be groups, such that we have $A \xrightarrow{f} B \xrightarrow{g} C$. This is called an exact sequence iff $\mathrm{im}(f) = \ker(g)$. An exact sequence is a short exact sequence iff it is of the form:
\[ 1 \to A \to B \to C \to 1 \]
\end{definition}

\section{Groups of order 8}

\begin{theorem}[Proposition 1.14]
The only groups of order 8 are $\mathbb{Z}/2\mathbb{Z} \times \mathbb{Z}/2\mathbb{Z} \times \mathbb{Z}/2\mathbb{Z}$, $\mathbb{Z}/2\mathbb{Z} \times \mathbb{Z}/4\mathbb{Z}$, $\mathbb{Z}/8\mathbb{Z}$, the quaternion group $Q$, and the dihedral group $D_8$.
\end{theorem}

\begin{theorem}[Theorem 1.7, Burnside's lemma]
Suppose there is some finite group $G$ which acts on a set $S$. Then the number of orbits is equal to the average number of fixed points.
\[ |S/G| = \frac{1}{|G|} \sum_{g \in G} |S^g| \]
where $S/G$ denotes the collection of orbits, and $S^g$ is the set of elements of $S$ fixed by $g$.
\end{theorem}

\begin{definition}[Definition 1.20, Conjugate elements]
Let $G$ be a group. Two elements $a, b \in G$ are conjugate iff there is some $g \in G$ such that $a = gbg^{-1}$.
\end{definition}

\begin{definition}[Definition 1.21, Quaternion group]
The quaternion group consists of 8 elements: $\pm 1, \pm i, \pm j, \pm k$ where
\[ ij = -ji = k, \quad ji = -ik = j, \quad jk = -kj = i, \quad i^2 = j^2 = k^2 = -1 \]
\end{definition}

\begin{definition}[Definition 1.22, Quaternion numbers]
The quaternion numbers are given by
\[ \{a + bI + cJ + dK \mid a, b, c, d \in \mathbb{R}\} \]
where we have a notion of conjugation given by:
\[ z = a + bI + cJ + dK, \quad \bar{z} = a - bI - cJ - dK \]
\end{definition}

\begin{theorem}[Proposition 1.17]
The quaternion numbers form a 4-dimensional division algebra containing $\mathbb{C}$. The non-zero quaternions form a group under multiplication.
\end{theorem}

\begin{definition}[Definition 1.23, Special orthogonal group]
The set of matrices $\mathrm{SO}_3(\mathbb{R})$ is the special orthogonal group iff it consists of all rotations in $\mathbb{R}^3$. Explicitly these are transformations that preserve the origin, euclidean distance, and orientation.
\end{definition}

\begin{theorem}[Proposition 1.20]
Conjugation by an element of $\mathbb{H}^\times$ gives a homomorphism $S^3 \to \mathrm{SO}_3(\mathbb{R})$.
\end{theorem}

All groups of order $p^n$ are nilpotent. Furthermore all products of groups of prime order are nilpotent.

\begin{definition}[Definition 1.26, Solvable group]
Let $G$ be a group. Then it is solvable iff it is either cyclic, or it has a normal subgroup $N$ such that $G/N$ is solvable.
\end{definition}

\begin{definition}[Definition 1.27, Simple group]
A group $G$ is simple iff it has no non-trivial normal subgroups.
\end{definition}

\begin{theorem}[Theorem 1.10, Sylow theorems]
Suppose $p^n$ divides $|G|$ and $p^{n+1}$ does not divide $|G|$. Then we have that:
1. There exists a subgroup of order $p^n$.
2. All Sylow subgroups are conjugate.
3. The number of $p$-Sylow subgroups of $G$ is $\equiv 1 \pmod{p}$.
4. Any subgroup of order $p^m$ for $m \leq n$ is contained in some $p$-Sylow subgroup.
\end{theorem}

\begin{theorem}[Theorem 1.11, Jordan-H\"older theorem]
Let $G$ be a group with a normal decomposition which eventually reaches the trivial group. Then the simple groups we get are independent of splitting choice.
\end{theorem}

\begin{theorem}[Theorem 1.12]
If $G$ is a group of order $pq$ for $p, q \in \mathbb{Z}$ primes, then if $p$ does not divide $(q - 1)$ we have that $G \cong \mathbb{Z}/pq\mathbb{Z}$. If $p$ does divide $q - 1$, we have that either $G \cong \mathbb{Z}/pq\mathbb{Z}$ or $G \cong \mathbb{Z}/q\mathbb{Z} \rtimes \mathbb{Z}/p\mathbb{Z}$.
\end{theorem}

2. $\mathbb{Z}/8\mathbb{Z} \times \mathbb{Z}/2\mathbb{Z}$
3. $\mathbb{Z}/4\mathbb{Z} \times \mathbb{Z}/4\mathbb{Z}$
4. $\mathbb{Z}/4\mathbb{Z} \times \mathbb{Z}/2\mathbb{Z} \times \mathbb{Z}/2\mathbb{Z}$
5. $(\mathbb{Z}/2\mathbb{Z})^4$
\end{theorem}

\section{Symmetric Group}

\begin{definition}[Definition 1.28, Symmetric group]
The symmetric group $S_n$ is the group of all permutations of $n$ elements.
\end{definition}

\begin{theorem}[Proposition 1.37]
Let $S_n$ be as in definition 1.28. Then $|S_n| = n!$.
\end{theorem}

\begin{definition}[Definition 1.29, Cycle]
A cycle $(a\; b\; c)$ is a function bringing $a \mapsto b \mapsto c \mapsto a$.
\end{definition}

\begin{definition}[Definition 1.30, Transposition]
A cycle is a transposition iff it exchanges 2 elements and leaves all the others.
\end{definition}

\begin{theorem}[Proposition 1.38]
The transpositions generate $S_n$. In other words, we can get any permutation by swapping adjacent elements.
\end{theorem}

\begin{definition}[Definition 1.31, Alternating group]
The alternating group is defined as follows:
\[ A_n := \{\sigma \in S_n : \varepsilon(\sigma) = 1\} \]
where $\varepsilon : S_n \to \{\pm 1\}$ is the sign homomorphism defined by the action on the discriminant $\Delta = \prod_{i < j}(x_i - x_j)$. Note $A_n = \ker(\varepsilon)$.
\end{definition}

\begin{theorem}[Proposition 1.39]
The group $A_n$ is normal in $S_n$ and of order $n!/2$ for $n \geq 2$.
\end{theorem}

\begin{definition}[Definition 1.32, Cycle shape]
The cycle shape of an element of $S_n$ is the multiplicity of cycles of each given size.
\end{definition}

\begin{theorem}[Theorem 1.14]
$A_n$ is simple for all $n \geq 5$.
\end{theorem}

\begin{definition}[Definition 1.33, Inner automorphism]
Let $G$ be a group. An inner automorphism of $G$ is given by the conjugate action of some element of $G$. This yields the exact sequence:
\[ 1 \to Z \to G \xrightarrow{\text{conjugations}} \mathrm{Aut}(G) \to \mathrm{Out}(G) \to 1 \]
where $\mathrm{Out}$ denotes the set of outer automorphisms which is the quotient $\mathrm{Aut}/\mathrm{Inn}$ where $\mathrm{Inn}$ is the subgroup consisting of all inner automorphisms.
\end{definition}

\begin{theorem}[Proposition 1.40]
$\mathrm{Aut}(S_n) \cong S_n$ for $n \geq 3$ except when $n = 6$. All of these automorphisms are inner automorphisms.
\end{theorem}

\section{Category Theory}

\begin{definition}[Definition 1.34, Category]
A category $\mathcal{C}$ is a collection of objects $\mathrm{Obj}(\mathcal{C})$ such that for any $A, B \in \mathrm{Obj}(\mathcal{C})$, there is a collection $\mathrm{Hom}(A, B)$ of maps (or arrows or morphisms) from $A$ to $B$. Also, for each $A, B, C \in \mathrm{Obj}(\mathcal{C})$ we have a function
\[ \mathrm{Hom}(B, C) \times \mathrm{Hom}(A, B) \to \mathrm{Hom}(A, C) \]
where $(g, f) \mapsto g \circ f$ called composition. Furthermore, for each $A \in \mathrm{Obj}(\mathcal{C})$, we have some element $\mathrm{id}_A \in \mathrm{Hom}(A, A)$ called the identity on $A$ such that:
1. Associativity: For each $f \in \mathrm{Hom}(A, B)$, $g \in \mathrm{Hom}(B, C)$ and $h \in \mathrm{Hom}(C, D)$ we have $(h \circ g) \circ f = h \circ (g \circ f)$.
2. Identity laws: For all $A, B \in \mathrm{Obj}(\mathcal{C})$, for all $f \in \mathrm{Hom}(A, B)$, we have $f \circ \mathrm{id}_A = f = \mathrm{id}_B \circ f$.
\end{definition}

\begin{definition}[Definition 1.35, Dual category]
If $\mathcal{C}$ is a category, the dual category is given by simply reversing all of the arrows. This is also written $\mathcal{C}^{\mathrm{op}}$.
\end{definition}

\begin{definition}[Definition 1.36, Functor]
Given two categories $\mathcal{A}$ and $\mathcal{B}$, a covariant functor between $\mathcal{A}$ and $\mathcal{B}$ is a function $F : \mathrm{Obj}(\mathcal{A}) \to \mathrm{Obj}(\mathcal{B})$ such that for all $A, A' \in \mathrm{Obj}(\mathcal{A})$ we have a function $F : \mathrm{Hom}(A, A') \to \mathrm{Hom}(F(A), F(A'))$ written $f \mapsto F(f)$ which satisfies the following axioms:
1. Whenever $A \xrightarrow{f} A' \xrightarrow{g} A''$ we have that $F(g \circ f) = F(g) \circ F(f)$.
2. $F(\mathrm{id}_A) = \mathrm{id}_{F(A)}$.
A contravariant functor $G$ from $\mathcal{A}$ to $\mathcal{B}$ is a covariant functor from $\mathcal{A}^{\mathrm{op}} \to \mathcal{B}$.
\end{definition}

\begin{definition}[Definition 1.37, Natural transformation]
Consider two categories $\mathcal{C}, \mathcal{D}$ and functors $F, G : \mathcal{C} \to \mathcal{D}$. A natural transformation is a map $\varphi : F \to G$ such that for each $a \in \mathrm{Obj}(\mathcal{C})$ we have $\varphi(a) : F(a) \to G(a)$ and the following diagram commutes: for any $f : a \to b$ in $\mathcal{C}$,
\[ G(f) \circ \varphi(a) = \varphi(b) \circ F(f) \]
\end{definition}

\begin{definition}[Definition 1.38, Product in a category]
Given a category $\mathcal{C}$, two elements $A, B \in \mathrm{Obj}(\mathcal{C})$ have a product $A \times B$ where $A \times B \in \mathrm{Obj}(\mathcal{C})$ equipped with projections $\pi_1 : A \times B \to A$ and $\pi_2 : A \times B \to B$ such that for any $X \in \mathrm{Obj}(\mathcal{C})$ with morphisms $f : X \to A$ and $g : X \to B$, there is a unique morphism $\varphi : X \to A \times B$ making the diagram commute. This defines $A \times B$ up to canonical isomorphism.
\end{definition}

\begin{definition}[Definition 1.39, Equalizer]
Let $\mathcal{C}$ be a category, and $A, B \in \mathrm{Obj}(\mathcal{C})$ with $f, g \in \mathrm{Hom}(A, B)$. Then the equalizer of $A, B$ is an object $E \in \mathrm{Obj}(\mathcal{C})$ and a morphism $i \in \mathrm{Hom}(E, A)$ such that $g \circ i = f \circ i$ and for any $C \in \mathrm{Obj}(\mathcal{C})$ such that there is some $h \in \mathrm{Hom}(C, A)$ where $g \circ h = f \circ h$, then $C$ factors uniquely through $E$.
\end{definition}

\begin{definition}[Definition 1.40, Kernel]
Suppose $A, B$ are groups, $f : A \to B$ is a group homomorphism, and $g : A \to B$ is the trivial homomorphism. Then $\ker(f)$ is the equalizer of $f$ and $g$.
\end{definition}

\begin{definition}[Definition 1.41, Initial object]
Let $A \in \mathrm{Obj}(\mathcal{C})$ for some category $\mathcal{C}$. Then $A$ is an initial object iff there exists a unique morphism from $A$ to any other element of $\mathrm{Obj}(\mathcal{C})$.
\end{definition}

\begin{theorem}[Proposition 1.41]
Initial objects are unique up to isomorphism.
\end{theorem}

\begin{definition}[Definition 1.42, Final object]
Let $A \in \mathrm{Obj}(\mathcal{C})$ for some category $\mathcal{C}$. Then $A$ is a final object iff there exists a unique morphism from any other element of $\mathrm{Obj}(\mathcal{C})$ to $A$.
\end{definition}

\begin{theorem}[Proposition 1.42]
Final objects are unique up to isomorphism.
\end{theorem}

\begin{definition}[Definition 1.43, Limit]
Let $\mathcal{C}$ be a category, $\{X_\alpha\}_{\alpha \in I}$ be a family of objects of $\mathcal{C}$, and $\{f_{\alpha_1, \alpha_2}\}_{\alpha_1, \alpha_2 \in I}$ be a family of morphisms where $f_{\alpha_1, \alpha_2} \in \mathrm{Hom}(X_{\alpha_1}, X_{\alpha_2})$. Then the limit of this family of objects is an object $A \in \mathrm{Obj}(\mathcal{C})$ and a collection of morphisms $\pi_\alpha \in \mathrm{Hom}(A, X_\alpha)$ such that:
\begin{itemize}
\item For any $f_{\alpha_1, \alpha_2}$ we have $\pi_{\alpha_2} = f_{\alpha_1, \alpha_2} \circ \pi_{\alpha_1}$.
\item Any other $B$ with these properties factors uniquely through $A$.
\end{itemize}
\end{definition}

\begin{definition}[Definition 1.44, Pullback]
Let $A, B, C \in \mathrm{Obj}(\mathcal{C})$ and $f : A \to C$, $g : B \to C$. Then $X \in \mathrm{Obj}(\mathcal{C})$ is called the pullback iff it is the limit of $A, B, C$ with morphisms $f, g$.
\end{definition}

\begin{definition}[Definition 1.45, Coproduct]
The coproduct is the product with all arrows reversed. In other words, the coproduct of $A, B \in \mathrm{Obj}(\mathcal{C})$ is some $A \amalg B \in \mathrm{Obj}(\mathcal{C})$ such that for any $X$ with morphisms $\varphi_1 : A \to X$ and $\varphi_2 : B \to X$, there is a unique morphism $\psi : A \amalg B \to X$ making the diagram commute.
\end{definition}

\section{Free groups}

\begin{definition}[Definition 1.46, Free abelian group]
The free abelian group on $n$ generators $\{g_i\}_{i=1}^n$ is the group of elements of the form
\[ \sum_{i=1}^n n_i g_i \]
where $n_i \in \mathbb{Z}$ for all $i$.
\end{definition}

\begin{theorem}[Proposition 1.43]
There is a unique isomorphism $\mathbb{Z}^n \cong G$ for any free abelian group $G$. Explicitly this brings $e_i \to g_i$. The free abelian group is the coproduct of $n$ copies of $\mathbb{Z}$.
\end{theorem}

\begin{definition}[Definition 1.47, Rank]
Let $G \cong \mathbb{Z}^n$ be a free abelian group. Then the exponent $n$ is the rank of $G$.
\end{definition}

\begin{theorem}[Proposition 1.44]
Any subgroup of $\mathbb{Z}^n$ is free of rank $\leq n$.
\end{theorem}

\begin{definition}[Definition 1.48, Free group]
The free group on generators $\{g_i\}_{i=0}^n$ is the universal group generated by these elements. It maps uniquely to any other group generated by $G_1, \ldots, G_n$ by $g_i \to G_i$. This is the coproduct of $\mathbb{Z}, \ldots, \mathbb{Z}$ in $\mathrm{Grp}$ rather than $\mathrm{Ab}$.
\end{definition}

\begin{theorem}[Theorem 1.15]
If $A$ is a reduced word (not the empty string) in a free group, we can find a finite group $G$ and a map from the free group to $G$ such that the image of $A$ is not 1.
\end{theorem}

\begin{definition}[Definition 1.49, Homotopy class]
Homotopy classes mean that two paths are equivalent if the difference between the two are just edges traversed that were immediately retraced backwards.
\end{definition}

\begin{definition}[Definition 1.50, Fundamental group of a graph]
The fundamental group of a graph consists of the homotopy classes of loops from the base point back to itself where loops follow the edges of the graph.
\end{definition}

\begin{theorem}[Proposition 1.45]
The fundamental group of a graph consisting of a single vertex mapping back to itself $n$ times is the free group on $n$ generators. Conversely, the fundamental group of any graph is free.
\end{theorem}
