\chapter[Algebraic extensions]{Algebraic extensions}

\section{Field extensions}

\begin{definition}[Definition 5.1, Field extension]
Let $K$ be a field. A field $L$ is a field extension of $K$ iff $K$ is a subfield of $L$, written $K \subseteq L$ or $L/K$.
\end{definition}

\begin{definition}[Definition 5.2, Degree]
Let $L$ be an extension of $K$. Then the degree $[L : K]$ of $L/K$ is $\dim L$ as a vector space over $K$.
\end{definition}

\begin{definition}[Definition 5.3, Algebraic element]
An element $\alpha \in L$ is algebraic over $K$ iff it is the root of a polynomial in $K[x]$.
\end{definition}

\begin{theorem}[Proposition 5.1]
Let $p$ be an irreducible polynomial. Then $K[x]/(p(x))$ is a field.
\end{theorem}

\begin{proof}
Consider the ideal generated by $p$. Since $p$ is irreducible, this ideal is maximal among the principal ideals of $K[x]$. Since $K[x]$ is a PID, this means $(p)$ is maximal in general, and therefore $K[x]/(p)$ is a field.
\end{proof}

\begin{theorem}[Theorem 5.1]
$\alpha$ is algebraic over $K$ iff it is contained in some finite extension $K_1$ of $K$, so we can write $[K_1 : K] < \infty$.
\end{theorem}

\begin{proof}
Let $\alpha \in K_1$ such that $[K_1 : K] = n < \infty$. Then the powers $1, \alpha, \alpha^2, \ldots, \alpha^n$ form a collection of $n + 1$ elements from an $n$-dimensional vector space over $K$, meaning we can write $a_0 + a_1\alpha + a_2\alpha^2 + \cdots + a_n\alpha^n = 0$ for some $\{a_j\} \subseteq K$ not all zero, so $\alpha$ is algebraic.
Now suppose $\alpha \in L$ is algebraic over $K$. Then there is some $p \in K[x]$ such that $p(\alpha) = 0$. WLOG take $p$ to be irreducible, meaning $K_1 := K[x]/(p)$ is a field. Then $[K_1 : K] = \deg(p)$ with basis $\{1, x, \ldots, x^{\deg(p)-1}\}$. This gives a map $\varphi : K[x]/(p) \to L$ where $x \mapsto \alpha$. Since $\varphi$ is injective and $K_1$ is a field, $\mathrm{im}(\varphi)$ is a field of finite degree containing $\alpha$.
\end{proof}

\begin{lemma}[Lemma 5.1, Tower law]
Let $K \subseteq K_1 \subseteq K_2$. Then we have $[K_2 : K] = [K_2 : K_1][K_1 : K]$.
\end{lemma}

\begin{proof}
Take $\{x_1, \ldots, x_m\}$ to be a basis of $K_1$ over $K$. Take $\{y_1, \ldots, y_n\}$ to be a basis of $K_2$ over $K_1$. Then $x_i y_j$ forms a basis of $K_2$ over $K$, so $[K_2 : K] = mn$ as desired.
\end{proof}

\begin{theorem}[Proposition 5.2]
Let $\alpha, \beta \in L$ be algebraic over $K$. Then $\alpha + \beta$ and $\alpha\beta$ are also algebraic over $K$.
\end{theorem}

\begin{proof}
Take $\alpha \in K_1$ and $[K_1 : K] < \infty$. We know that $\beta$ is the root of some irreducible polynomial of finite degree $n$ over $K_1$, so $\beta$ is algebraic over $K_1$. Let $\beta \in K_2$ where $[K_2 : K_1] < \infty$. Then from the tower law, $[K_2 : K] = [K_2 : K_1][K_1 : K] < \infty$, so since $\alpha\beta, \alpha + \beta \in K_2$ we have the desired result.
\end{proof}

\begin{theorem}[Proposition 5.3]
Let $p \in K[x]$ be irreducible. We can find an extension field $L \supseteq K$ such that $p$ has a root in $L$.
\end{theorem}

\begin{proof}
Take $L = K[x]/(p)$.
\end{proof}

\section{Splitting fields}

\begin{definition}[Definition 5.4, Splitting field]
Let $p \in K[x]$ with $K \subseteq L$. Then $L$ is a splitting field of $p$ iff:
1. $p$ factors linearly in $L$.
2. $L$ is generated by roots of $p$.
\end{definition}

\begin{theorem}[Proposition 5.4]
Let $L_1, L_2$ be two splitting fields of a field $K$. Then we can find an isomorphism from $L_1$ to $L_2$ fixing all elements of $K$. This isomorphism is not necessarily unique.
\end{theorem}

\begin{proof}
Construct a sequence of splitting fields $K \subseteq K_1 \subseteq K_2 \cdots \subseteq K_n$. Let $L$ be a splitting field of $K$. Then we get maps $\varphi_i : K_i \to L$ since each $K_i = K_{i-1}[x]/(q_i)$ and $L$ is a splitting field of $p$. Since $\mathrm{im}(\varphi_n) = L$ (as $L$ is generated by the roots of $p$), we have $K_n \cong L$.
\end{proof}

\begin{theorem}[Proposition 5.5]
For any prime power $p^n$ there is a unique field $\mathbb{F}_{p^n}$ with $p^n$ elements.
\end{theorem}

\begin{proof}
$\mathbb{F}_{p^n}$ is the splitting field of the polynomial $x^{p^n} - x$. This polynomial has $p^n$ distinct roots because its derivative $px^{p^n-1} - 1$ is coprime to $x^{p^n} - x$. The roots form a field since $(x + y)^p = x^p + y^p$ and the roots are either 0 or roots of $x^{p^n - 1} = 1$.
For uniqueness, any field of order $p^n$ is a splitting field of $x^{p^n} - x$. All elements must be roots: if $x = 0$ it is a root; if $x \neq 0$, then $x \in L^\times$ which has order $p^n - 1$, so $x^{p^n - 1} = 1$ by Lagrange's theorem.
\end{proof}

\section{Algebraic closure}

\begin{definition}[Definition 5.5, Algebraic closure]
$L \supseteq K$ is called the algebraic closure of $K$ iff:
1. Every element of $L$ is algebraic over $K$.
2. Every polynomial in $L[x]$ has a root.
\end{definition}

\begin{theorem}[Proposition 5.6]
Let $K$ be a field. Then $K$ has an algebraic closure $L$ unique up to isomorphism.
\end{theorem}

\begin{proof}
Let there be some countable family of polynomials $\{p_i\}_{i \in \mathbb{N}}$. Form $K \subseteq K_1 \subseteq K_2 \subseteq \cdots$ where $K_i$ is a splitting field of $p_i$ over $K_{i-1}$. The union of the $K_i$ is a splitting field. For uncountably many polynomials, use Zorn's lemma. We find $L \supseteq K$ such that any polynomial in $K[x]$ has a root in $L$. If $p$ is irreducible in $L[x]$, consider $M := L[x]/(p)$. The coefficients of $p$ are in $K$ and therefore in some finite extension, so $\alpha$ is algebraic over $K$, meaning $\alpha \in L$.
\end{proof}

\section{Normal extensions}

\begin{theorem}[Proposition 5.7]
A field $L$ is a splitting field for some family of polynomials iff an irreducible polynomial having one root in $L$ implies it has all roots in $L$.
\end{theorem}

\begin{proof}
Let $p \in K[x]$ be irreducible with root $\alpha \in L$. Let $M$ be the algebraic closure of $L$. Any homomorphism $\varphi : K[\alpha] \to M$ extends to $\psi : L \to M$ since $M$ is algebraically closed. Furthermore, $\mathrm{im}(\psi) = L$ since $L$ is the splitting field of some family of polynomials. Therefore $\alpha$ is already in $L$.
\end{proof}

\begin{definition}[Definition 5.6, Normal extension]
A finite field extension $L/K$ is normal iff the existence of 1 root of an irreducible polynomial $p$ implies the existence of all the roots of $p$ in $L$.
\end{definition}

\begin{theorem}[Proposition 5.8]
An extension $L/K$ is normal iff it is the splitting field of some family of polynomials.
\end{theorem}

\begin{theorem}[Proposition 5.10]
Let $L/K$ be a field extension. If $[L : K] = 2$ then $L$ is normal.
\end{theorem}

\begin{proof}
Let $\alpha$ be a root of a polynomial $a^2 + ax + b = (a - \alpha)(a - \beta)$. Then $\alpha + \beta = -a$ meaning $\beta = -a - \alpha$ so $\beta \in K[\alpha]$ as desired.
\end{proof}

\section{Separable extensions}

\begin{definition}[Definition 5.7, Separable polynomial]
A polynomial $p$ is called separable iff it has no repeated roots.
\end{definition}

\begin{theorem}[Proposition 5.11]
Let $p$ be a polynomial. Then $p$ is separable iff $p, p'$ are coprime.
\end{theorem}

\begin{definition}[Definition 5.8, Separable element]
Let $L/K$ be a field extension. $\alpha \in L$ is separable iff its irreducible polynomial is separable.
\end{definition}

\begin{definition}[Definition 5.9, Separable extension]
A field extension $L/K$ is separable iff all of the elements are separable.
\end{definition}

\begin{theorem}[Theorem 5.2]
Let $L/K$ be a field extension. If $K$ has characteristic 0, then $L/K$ is separable.
\end{theorem}

\begin{proof}
Let $\alpha$ be a root of some irreducible polynomial $p$. Then $\deg(p) > \deg(p')$ so $p, p'$ have no common factors since $p$ is irreducible. Therefore $p, p'$ are coprime.
\end{proof}

\begin{corollary}[Corollary 5.1]
Any extension $\mathbb{F}_q / \mathbb{F}_p$ of finite fields is separable.
\end{corollary}

\begin{proof}
Any element is a root of the polynomial $p(x) = x^q - x$ which has derivative $p'(x) = -1$ so $(p, p') = 1$.
\end{proof}

\section{Galois extensions}

\begin{definition}[Definition 5.10, Galois extension]
A field extension is Galois iff it is separable and normal.
\end{definition}

\begin{definition}[Definition 5.11, Galois group]
Let $L/K$ be a Galois extension. The Galois group of $L/K$ written $\mathrm{Gal}(L/K)$ is the group of automorphisms of $L$ fixing $K$.
\end{definition}

\begin{lemma}[Lemma 5.2]
Let $L/K$ be a finite extension of degree $n$. Then there are at most $n$ ways to get a map $f : L \to M$ extending the inclusion of $K$.
\end{lemma}

\begin{proof}
Suppose $L$ is generated by $\alpha$ so $L = K[\alpha]$. Then $\alpha$ is a root of some polynomial $p$ of degree $\leq n$. $f(\alpha)$ is also such a root in $M$ but $p$ has at most $n$ roots.
\end{proof}

\begin{theorem}[Theorem 5.3]
Let $L/K$ be a finite extension. Then the following are equivalent:
1. $L$ is a splitting field of a separable polynomial.
2. $L$ is Galois.
3. $[L : K] = |G|$ where $G$ is the group of automorphisms of $L$ fixing $K$.
4. $K = L^G$ is the elements of $L$ fixed by $G$.
\end{theorem}

\begin{proof}
$(1) \implies (2)$: All splitting fields are normal.

$(2) \implies (3)$: Take $K \subseteq L \subseteq M$ where $M$ is the algebraic closure. $L/K$ is separable, so there are $[L : K] = n$ extension maps. $L/K$ is normal, so the image of any map from $L$ to $M$ lies in $L$. From Lemma 5.2, there are $\leq [L : K]$ maps $L \to L$, so $|G| = [L : K]$.

$(3) \implies (4)$: Consider $K \subseteq L^G \subseteq L$. There are $\geq n$ automorphisms extending $L^G$. Therefore $[L : L^G] \geq n$, but $[L : K] = n$ so $K = L^G$.

$(4) \implies (1)$: Let $\alpha \in L$ and consider the conjugates under $G$. Look at $(x - \alpha)(x - \beta)(x - \gamma) \cdots$. This is in $K[x]$ since all coefficients are invariant under $G$ (as $K = L^G$). So $\alpha$ is a root of a separable polynomial, and it splits into linear factors, giving normality.
\end{proof}

\section{Main theorem of Galois theory}

\begin{theorem}[Theorem 5.4, Fundamental theorem of Galois theory]
Let $M/K$ be a Galois extension and $G = \mathrm{Gal}(M/K)$. We have a correspondence between sub-extensions $L$ where $K \subseteq L \subseteq M$ and subgroups $H \subseteq G$. Explicitly we send
\[ L \mapsto \mathrm{Gal}(M/L), \quad M^H \leftarrow H \]
where $M^H$ is the set of elements of $M$ fixed by $H$.
\end{theorem}

\begin{proof}
We want to show these embeddings are equalities. It is enough to show they have the same ``size.''
(1): $|\mathrm{Gal}(M/L)| = [M : L]$. We know $|\mathrm{Gal}(M/L)| \leq [M : L]$. If the inequality is strict, there are $\leq [L : K]$ extension maps from $L$ to $M$ and $< [M : L]$ maps from $M$ to $M$, giving fewer than $[L : K][M : L] = [M : K]$ maps from $M$ to $M$. But since $M/K$ is Galois there are $[M : K]$ such maps, a contradiction.
(2): $[M : M^H] = |H|$ is left as an exercise.
\end{proof}

\section{Normal extensions and normal subgroups}

\begin{theorem}[Proposition 5.12]
Let $H \subseteq \mathrm{Gal}(L/K)$. Then $H$ is normal iff $L^H/K$ is a normal extension.
\end{theorem}

\begin{proof}
$H \subseteq G = \mathrm{Gal}(L/K)$ is normal iff all conjugates of $H$ under $G$ equal $H$. $M/K$ is normal iff all conjugates of $M$ under $G$ equal $M$. The result follows.
\end{proof}

\begin{theorem}[Proposition 5.13]
Let $G$ be a finite group. Then there is some Galois extension $L/K$ such that $\mathrm{Gal}(L/K) = G$.
\end{theorem}

\begin{proof}
Take $G = S_n$, and $L = \mathbb{Q}(x_1, \ldots, x_n)$ the rational functions on $n$ variables. Let $S_n$ act on $L$, and put $K = L^{S_n}$ the symmetric rational functions. If $G$ is any finite group acting on any field $L$, then $L/L^G$ is Galois with $\mathrm{Gal}(L/L^G) = G$. The same works for $G \subseteq S_n$ via Cayley's theorem.
\end{proof}

\begin{theorem}[Proposition 5.14]
Let $p \in \mathbb{Z}$ be prime. Then we can find an extension $L/\mathbb{Q}$ such that $\mathrm{Gal}(L/\mathbb{Q}) = S_p$.
\end{theorem}

\begin{proof}
We need to find an irreducible polynomial of degree $p$ with exactly two complex roots.
\end{proof}

\begin{corollary}[Corollary 5.2]
Let $G$ be a finite group. We can find an extension $L/K$ of $\mathbb{Q}$ such that $\mathrm{Gal}(L/K) = G$.
\end{corollary}

\begin{proof}
Let $L$ be the extension such that $\mathrm{Gal}(L/\mathbb{Q}) = S_p$ for some sufficiently large $p$ such that $G \subseteq S_p$. Then take $K = L^G$.
\end{proof}

\section{Applications to polynomials}

\begin{theorem}[Proposition 5.15]
Consider an irreducible polynomial $p(x) = x^3 + ax^2 + bx + c = 0$ in $K[x]$. Then the Galois group $G$ of the splitting field is $S_3$ iff the discriminant of $p$ is not a square in $K$. If the discriminant is a square, then the Galois group is cyclic of order 3, equal to $A_3$.
\end{theorem}

\begin{proof}
The Galois group $G \subseteq S_3$. Look at $\Delta = (\alpha - \beta)(\beta - \gamma)(\gamma - \alpha)$ where $\alpha, \beta, \gamma$ are the roots. $\Delta$ is fixed by even permutations but changes sign under odd permutations. If $G = \mathbb{Z}/3\mathbb{Z}$ then $\Delta \in K$. If $G = S_3$ then $\Delta \mapsto -\Delta$ must be an automorphism. Therefore $\Delta^2$ has a square root in $K$ iff $G = A_3$. Explicitly, $\Delta^2 = -4b^3 - 27c^2$ if $a = 0$.
\end{proof}

\begin{theorem}[Theorem 5.5, Fundamental theorem of algebra]
$\mathbb{C}$ is algebraically closed.
\end{theorem}

\begin{proof}
We use the following facts about $\mathbb{R}$ and $\mathbb{C}$: (1) $\mathbb{R}$ has characteristic 0; (2) Any odd degree polynomial in $\mathbb{R}[x]$ has a real root; (3) $[\mathbb{C} : \mathbb{R}] = 2$ and every element of $\mathbb{C}$ has a square root. Let $L$ be a finite extension of $\mathbb{C}$. We may assume $L$ is a Galois extension of $\mathbb{R}$ (since every algebraic extension of a field of characteristic 0 is separable). Let $G = \mathrm{Gal}(L/\mathbb{R})$. Item (2) gives $G$ has no subgroups of odd index $> 1$. Item (3) gives no subgroups of $\mathbb{C}$ have index 2. Take any 2-Sylow subgroup $S$ of $G$; $S$ has odd index so $S = G$. So $|G| = 2^n$. Then $|H| = 2^{n-1}$. If $n - 1 > 0$, $H$ has subgroups of index 2, contradicting item (3). So $|H| = 1$ and $|G| = 2$, meaning $L = \mathbb{C}$.
\end{proof}

\begin{lemma}[Lemma 5.3]
Let $V$ be a vector space over an infinite field $K$. Then $V$ is not a union of finitely many proper subspaces of $V$.
\end{lemma}

\begin{proof}
Proceed by induction. Let $V_1, \ldots, V_n$ be proper subspaces. Take $v \notin V_1, \ldots, V_{n-1}$ by induction and $w \notin V_n$. Then $u = v + kw$ for $k \in K$. There is at most 1 value of $k$ for which $u \in V_i$ for any given $i$. Since $K$ is infinite, we can choose $k$ such that $v + kw$ is not in any $V_j$.
\end{proof}

\begin{theorem}[Theorem 5.6, Primitive element theorem]
Let $L/K$ be a finite separable extension. Then $L$ is generated by 1 element. In other words there exists some $\alpha \in L$ such that $L = K(\alpha)$.
\end{theorem}

\begin{proof}
There are only finitely many extensions between $K$ and $L$ (corresponding to subgroups of the Galois group of a Galois extension containing $L$). Each extension is a vector space over $K$. If $K$ is infinite, then $L$ is not a union of finitely many subspaces, so some $\alpha \in L$ is not in any smaller extension, meaning $L = K(\alpha)$. If $K$ is finite, then $L$ is finite, so $L^\times$ is cyclic.
\end{proof}

\section{Cyclic extensions}

\begin{definition}[Definition 5.12, Cyclic extension]
Let $K$ be a field, and $L/K$ be a Galois extension. $L/K$ is a cyclic extension iff $\mathrm{Gal}(L/K)$ is a cyclic group.
\end{definition}

\begin{theorem}[Proposition 5.16]
Let $K = \mathbb{Q}(\zeta)$ where $\zeta$ is a primitive $p$-th root of unity. Let $L = K(\sqrt[p]{a})$ for some $a \in K$. Then $\mathrm{Gal}(L/K) = \mathbb{Z}/p\mathbb{Z}$.
\end{theorem}

\begin{proof}
$\sqrt[p]{a}$ is a root of $x^p - a$. The other roots are $\sqrt[p]{a}, \sqrt[p]{a}\zeta, \sqrt[p]{a}\zeta^2, \ldots$. Any element of the Galois group takes $\sqrt[p]{a}$ to $\sqrt[p]{a}\zeta^i$ for some $i$. Since $p$ is prime, the Galois group is either trivial or $\mathbb{Z}/p\mathbb{Z}$ itself.
\end{proof}

\begin{theorem}[Proposition 5.17]
Let $K$ contain all $p$-th roots of unity, and let $\mathrm{char}(K) \neq p$. Then if $\mathrm{Gal}(L/K) = \mathbb{Z}/p\mathbb{Z}$ for some extension $L/K$, then $L = K(\sqrt[p]{a})$ for some $a \in K$.
\end{theorem}

\begin{proof}
Let $\sigma$ be a generator of $\mathbb{Z}/p\mathbb{Z}$. Look at $\sigma$ as a linear transformation on $L$ over $K$. Its eigenvalues are roots of $x^p - 1$ which lie in $K$. Find eigenvectors: $v + \zeta^{-1}\sigma v + \cdots + \zeta^{1-p}\sigma^{p-1}v$ has eigenvalue $\zeta$. Pick $w$ an eigenvector with eigenvalue $\neq 1$. Then $\sigma w = \zeta w$ so $\sigma w^p = \zeta^p w^p = w^p$. So $w^p \in K$. Take $a = w^p$ so $L = K(\sqrt[p]{a})$.
\end{proof}

\begin{theorem}[Theorem 5.7]
Let $L/K$ be a Galois extension. If $\mathrm{Gal}(L/K) = \mathbb{Z}/p\mathbb{Z}$, $K$ contains roots of $1 + x + \cdots + x^{p-1}$, and $\mathrm{char}(K) \neq p$, then $L = K(\sqrt[p]{a})$ for some $a \in K$.
\end{theorem}

\begin{theorem}[Proposition 5.18]
Let $\mathrm{char}(K) = p$, and $L/K$ be a Galois extension such that $[L : K] = p$. We have only two possibilities:
1. $x^p - x - b$ is irreducible, yielding a Galois extension with Galois group $\mathbb{Z}/p\mathbb{Z}$.
2. $x^p - x - b$ factors into linear factors, where $b = c^p - c$ for $c \in K$.
\end{theorem}

\begin{proof}
Let $\sigma$ generate the Galois group. Since $\sigma^p = 1$, $(\sigma - 1)^p = 0$ so $\sigma - 1$ is nilpotent. The only eigenvalue is 1 with eigenspace $K$. Find a generalized eigenvector: for some $a \in K$ and $v \in L$, $\sigma v - v = a$. Normalizing, $\sigma v = v + 1$. Then $\sigma v^p = v^p + 1$ so $v^p - v \in K$. Therefore $v$ is a root of $x^p - x - b$ for some $b \in K$ (an Artin-Schreier equation). The other roots are $v, v+1, \ldots, v+(p-1)$, which are $p$ distinct roots. So $K(v)$ is separable and normal, hence Galois.
\end{proof}

\begin{theorem}[Theorem 5.8]
The Galois group of the splitting field of a polynomial is solvable iff roots can be given using radicals and Artin-Schreier equations (in characteristic $> 0$).
\end{theorem}

\begin{proof}
Let an equation be solvable by radicals. Assume $K$ contains all required roots of unity. Look at $K_0 \subseteq K_1 \subseteq \cdots \subseteq L$ where $L$ is the splitting field. Then the corresponding groups $G \supseteq G_1 \supseteq \cdots \supseteq 1$ where $G_i$ is normal in $G_{i-1}$ and $G_{i-1}/G_i$ is cyclic of prime order. Each $K_{i+1}/K_i$ is a cyclic Galois extension, so $K_{i+1} = K_i(\sqrt[n]{a})$ or Artin-Schreier.
\end{proof}

\begin{theorem}[Proposition 5.19]
Let $K$ be a field such that $\mathrm{char}(K) = 0$. All polynomials of degree $\leq 4$ in $K[x]$ can be solved by radicals.
\end{theorem}

\begin{proof}
The Galois group is a subgroup of $S_4$ so it is solvable. We have $S_4 \supseteq A_4 \supseteq (\mathbb{Z}/2\mathbb{Z}) \times (\mathbb{Z}/2\mathbb{Z}) \supseteq 1$, all quotients are abelian.
\end{proof}

\section{Cyclotomic polynomials}

\begin{definition}[Definition 5.13, Cyclotomic polynomial]
The $n$-th cyclotomic polynomial, written $\Phi_n(x)$, is the polynomial with the primitive $n$-th roots as its roots.
\end{definition}

\begin{theorem}[Theorem 5.9]
$\Phi_n(x)$ is irreducible over $\mathbb{Q}$. The corresponding Galois group is $(\mathbb{Z}/n\mathbb{Z})^\times$.
\end{theorem}

\begin{proof}
Let $f$ be an irreducible factor of $\Phi_n(x)$ over $\mathbb{Q}$. Form $\mathbb{Z}[\zeta] = \mathbb{Z}[x]/f(x)$. This contains $n$ distinct roots of $x^n - 1$. Choose an irreducible factor $g(x)$ of $f(x)$ in $\mathbb{F}_p[x]$. Since $g$ is irreducible and $\zeta$ is a root, $\zeta^p$ is also a root of $g$ (since $t \mapsto t^p$ is an automorphism of $\mathbb{F}_p(\zeta)$). So in $\mathbb{Z}[\zeta]$, the roots of $f$ are closed under $\zeta \mapsto \zeta^p$ for $p \nmid n$. Automorphisms take $\zeta \mapsto \zeta^k$ for $k, n$ coprime, generating $(\mathbb{Z}/n\mathbb{Z})^\times$. So $f = \Phi_n(x)$.
\end{proof}

\begin{definition}[Definition 5.14, Cyclotomic field]
A cyclotomic field is a field generated by roots of unity.
\end{definition}

\begin{theorem}[Theorem 5.10]
Let $n \in \mathbb{Z}$. There are infinitely many primes $p > 0$ with $p \equiv 1 \pmod{n}$.
\end{theorem}

\begin{proof}
Look at primes $p$ dividing $\Phi_n(a)$ for some $a$. Let $p, n$ be coprime. If $p \mid \Phi_n(a)$, then $p \nmid \Phi_m(a)$ for $m \mid n$. So $a$ has order exactly $n$ modulo $p$, meaning $n \mid (p-1)$, so $p \equiv 1 \pmod{n}$. Given finitely many such primes $p_1, \ldots, p_k$, choose $p$ dividing $\Phi_n(np_1 \cdots p_k)$. Then $p$ is a new prime $\equiv 1 \pmod{n}$.
\end{proof}

\begin{theorem}[Theorem 5.11]
Let $G$ be a finite Abelian group. Then there exists some $K/\mathbb{Q}$ such that $G$ is the Galois group of $K/\mathbb{Q}$.
\end{theorem}

\begin{proof}
Write $G = (\mathbb{Z}/n_1\mathbb{Z}) \times (\mathbb{Z}/n_2\mathbb{Z}) \times \cdots$. Choose distinct primes $p_i \equiv 1 \pmod{n_i}$. Then $\mathbb{Z}/n_i\mathbb{Z}$ is a quotient of $(\mathbb{Z}/p_i\mathbb{Z})^\times$. So $G$ is a quotient of $(\mathbb{Z}/p_1 \cdots p_n\mathbb{Z})^\times$, which is the Galois group of $x^{p_1 \cdots p_n} - 1$. Take $K$ to be the fixed field of the kernel.
\end{proof}

\begin{theorem}[Theorem 5.12, Kronecker-Weber-Hilbert]
If $K/\mathbb{Q}$ is a Galois extension with Abelian Galois group, then $K \subseteq \mathbb{Q}[\zeta]$ for $\zeta$ some root of unity.
\end{theorem}

\begin{theorem}[Theorem 5.13, Wedderburn's theorem]
Any finite division algebra is a field, and is therefore commutative.
\end{theorem}

\begin{proof}
Let $L$ be a finite division algebra, $K$ its center, and $\mathbb{F}_q$ a field of order $q$. Consider $G = L^\times$ of order $q^n - 1$. The class equation gives $q^n - 1 = q - 1 + \sum_i \frac{q^n - 1}{q^{k_i} - 1}$ where $k_i < n$. Now $q^n - 1$ is divisible by $\Phi_n(q) = \prod_{i \in (\mathbb{Z}/n\mathbb{Z})^\times} (q - \zeta^i)$. Also $(q^n - 1)/(q^{k_i} - 1)$ is divisible by $\Phi_n(q)$ since $k_i < n$. So $q - 1$ is divisible by $\Phi_n(q)$. But $|q - \zeta^i| > q - 1$ unless $\zeta^i = 1$, meaning $n = 1$ and $L = K$.
\end{proof}

\begin{definition}[Definition 5.15, Brauer group]
The Brauer group is the group of isomorphism classes of finite dimensional division algebras over a field $K$ with center $K$.
\end{definition}

\section{Norm and trace}

\begin{definition}[Definition 5.16, Norm and trace]
Let $L/K$ be a finite extension. Taking $a \in L$, the trace of $a$ is its trace when viewed as a linear transformation from $L \to L$ where $L/K$ is a vector space over $K$. The norm of $a$, written $N_{L/K}(a)$, is the determinant of $a$ as a linear transformation in the same sense.
\end{definition}

\begin{theorem}[Proposition 5.21]
Let $a, b \in L/K$ be elements of a finite extension. Then
\[ N_{L/K}(ab) = N_{L/K}(a) N_{L/K}(b), \quad \mathrm{Tr}(a + b) = \mathrm{Tr}(a) + \mathrm{Tr}(b) \]
\end{theorem}

\begin{proof}
This follows directly from the corresponding statements for matrices.
\end{proof}

\begin{theorem}[Proposition 5.22]
Let $L = K[a]$ be an extension generated by $a$. Then $a$ satisfies some irreducible polynomial $x^n + b_{n-1}x^{n-1} + \cdots + b_0 = 0$. Denote the roots by $a_1, \ldots, a_n$. Then
\[ \mathrm{Tr}(a) = -b_{n-1} = \sum_{i=1}^n a_i, \quad N(a) = \pm b_0 = \prod_{i=1}^n a_i \]
\end{theorem}

\begin{proof}
Choose basis $\{1, a, a^2, \ldots, a^{n-1}\}$ for $L$ over $K$. The matrix representing $a$ is the companion matrix with trace $-b_{n-1}$ and determinant $\pm b_0$. The roots satisfy $b_{n-1} = a_1 + \cdots + a_n$ and $b_0 = \pm a_1 \cdots a_n$.
\end{proof}

\begin{theorem}[Theorem 5.14]
Suppose $K \subseteq K[a] \subseteq L$. Then $N_L(a)$ is the product of all roots of the irreducible polynomial of $a$, where we count each root $[L : K[a]]$ times. Likewise for $\mathrm{tr}(a)$.
\end{theorem}

\begin{theorem}[Proposition 5.23]
Let $L/K$ be a Galois extension, and $G = \mathrm{Gal}(L/K)$. Then
\[ N(a) = \prod_{\sigma \in G} \sigma(a), \quad \mathrm{tr}(a) = \sum_{\sigma \in G} \sigma(a) \]
\end{theorem}

\begin{proof}
If $a$ is some root, then the other roots are given by $\sigma_i(a)$ for $\sigma_i \in G$. The result follows.
\end{proof}

\begin{definition}[Definition 5.17, Algebraic integers]
The algebraic integers of a field $K$ are the roots of polynomials in $\mathbb{Z}[x]$ with leading coefficient 1.
\end{definition}

\begin{theorem}[Theorem 5.15]
Let $L/\mathbb{Q}$ be a finite extension. Then for $\alpha \in L$, the following are equivalent:
1. $\alpha$ is an algebraic integer (root of $x^n + \cdots = 0$).
2. There exists some finitely generated $\mathbb{Z}$-module $A$ in $L$ spanning $L$, so that $\alpha A \subseteq A$.
\end{theorem}

\begin{proof}
$(1) \implies (2)$: Take $A$ spanned by $\{1, \alpha, \alpha^2, \ldots, \alpha^{n-1}\}$. Then $\alpha \cdot \alpha^{n-1}$ is a linear combination of the spanning set.
$(2) \implies (1)$: Think of $\alpha$ as a linear transformation of the free $\mathbb{Z}$-module $A$. Then $\alpha$ is a root of its characteristic polynomial, which has leading coefficient 1.
\end{proof}

\begin{theorem}[Proposition 5.24]
The algebraic integers form a subring.
\end{theorem}

\begin{theorem}[Proposition 5.25]
If $\alpha$ is an algebraic integer of some field $K$, then $\mathrm{tr}(\alpha)$ and $N(\alpha)$ are also algebraic integers.
\end{theorem}

\begin{proof}
The conjugate of $\alpha$ is also algebraic, and the sum and product of algebraic integers are algebraic.
\end{proof}

\begin{theorem}[Proposition 5.26]
Let $L = \mathbb{Q}[\sqrt{N}]$ where $N \in \mathbb{Z}$ is square free. Then the algebraic integers are:
\[ \mathbb{Z}[\sqrt{N}] \quad \text{if } N \equiv 2, 3 \pmod{4} \]
\[ \mathbb{Z}\left[\frac{1 + \sqrt{N}}{2}\right] \quad \text{if } N \equiv 1 \pmod{4} \]
\end{theorem}

\begin{proof}
For algebraic integer $\alpha = m + n\sqrt{N}$, the matrix representation is $\begin{pmatrix} m & n \\ nN & m \end{pmatrix}$. The trace is $2m$ and the norm is $m^2 - n^2N$. If $m \in \mathbb{Z}$ and $m^2 - n^2N \in \mathbb{Z}$, then $n \in \mathbb{Z}$ since $N$ is square free. If $m \notin \mathbb{Z}$ (i.e., $m \in \mathbb{Z} + 1/2$), then $1/4 - n^2N \in \mathbb{Z}$ so $1 \equiv (2n)^2 N \pmod{4}$. If $N \equiv 2$ or $3 \pmod{4}$ there are no solutions. If $N \equiv 1 \pmod{4}$ there are solutions.
\end{proof}

\begin{theorem}[Proposition 5.27]
Let $L/K$ be a field extension. Then $\mathrm{tr}$ gives a bilinear form $(a, b) = \mathrm{tr}(ab)$ for all $a, b \in L$.
\end{theorem}

\begin{theorem}[Theorem 5.16]
If $L/K$ is a separable extension, then the bilinear form given by $\mathrm{tr}_{L/K}$ is non-degenerate.
\end{theorem}

\begin{definition}[Definition 5.18, Characters]
Let $G$ be a group, $K$ a field. The characters of $G$ are homomorphisms from $G$ to $K^\times$.
\end{definition}

\begin{lemma}[Lemma 5.4, Artin's lemma]
Let $G$ be a group. If $\chi_1, \ldots, \chi_n$ are distinct characters of $G$, they are linearly independent. In other words, if $a_1\chi_1(g) + a_2\chi_2(g) + \cdots + a_n\chi_n(g) = 0$ for all $g \in G$, then $a_1 = \cdots = a_n = 0$.
\end{lemma}

\begin{proof}
Proceed by contradiction. Suppose $a_i \neq 0$ for all $i$, and $n$ is minimal. Find $h \in G$ such that $\chi_1(h) \neq \chi_2(h)$. Then $a_1\chi_1(g)\chi_1(h) + \cdots + a_n\chi_n(g)\chi_n(h) = 0$. Subtracting $\chi_1(h)$ times the original equation: $a_2(\chi_1(h) - \chi_2(h))\chi_2(g) + \cdots = 0$, contradicting minimality of $n$.
\end{proof}

\begin{definition}[Definition 5.19, Discriminant of a field extension]
Let $L/K$ be a field extension, and $\{a_i\}_{i=1}^n$ a basis for $L$ over $K$. The discriminant of the extension is $\det((a_i, a_j))$ where $(a_i, a_j) = \mathrm{tr}(a_i a_j)$.
\end{definition}

\begin{theorem}[Theorem 5.17]
Let $L/K$ be a finite extension of a finite field. Then $N : L^\times \to K^\times$ is onto.
\end{theorem}

\begin{proof}
$G = \mathrm{Gal}(L/K)$ is cyclic, generated by Frobenius $F : x \to x^q$ for $q = |K|$. Then $N(a) = a \cdot Fa \cdots F^{n-1}a = a^{(q^n - 1)/(q-1)}$. There are at most $(q^n - 1)/(q-1)$ elements of norm 1. Since $|\ker(N)| \times |\mathrm{im}(N)| = |L^\times| = q^n - 1$ and $|\mathrm{im}(N)| \leq q - 1$, these are equalities, so $N$ is onto.
\end{proof}

\begin{theorem}[Theorem 5.18, Hilbert's theorem 90]
Let $L/K$ be a cyclic Galois extension, and let $\sigma$ generate $G = \mathrm{Gal}(L/K)$. Then $N(a) = 1$ for $a \in L$ iff there is some $b \in L^\times$ such that $a = b/\sigma b$.
\end{theorem}

\begin{proof}
$(\Leftarrow)$: If $a = b/\sigma b$ then $N(a) = 1$ trivially.
$(\Rightarrow)$: We need to solve $a\sigma b = b$ for $b \neq 0$. Note $(a\sigma)^n = a \cdot \sigma(a) \cdots \sigma^{n-1}(a) \cdot \sigma^n = N(a) \cdot 1 = 1$. A fixed point of any group $G$ can be given by $\sum_{g \in G} g(v)$. Take $b = \sum_{i \in \mathbb{Z}/n\mathbb{Z}} (a\sigma)^i(\theta)$ for some $\theta \in L$. This $b$ is fixed by $a\sigma$. By Artin's lemma, the $\sigma^i$ are linearly independent, so we can choose $\theta$ with $b \neq 0$.
\end{proof}

\begin{theorem}[Theorem 5.19, Generalized Hilbert's theorem 90]
For any Galois extension $L/K$, $H^1(G, L^\times) = 1$ where $G = \mathrm{Gal}(L/K)$.
\end{theorem}

\begin{proof}
Suppose $a_\sigma \in L^\times$ with $a_{\sigma\tau} = a_\sigma \cdot \sigma a_\tau$. Look at $\sigma \mapsto a_\sigma \cdot \sigma$ as a linear map $L \to L$. We want $b$ such that $a_\sigma \sigma b = b$ for all $\sigma$, $b \neq 0$. Take $b = \sum_{\sigma \in G} a_\sigma \cdot \sigma v$. By Artin's lemma, the $\sigma$ are linearly independent, so we can find $v$ with $b \neq 0$.
\end{proof}

\begin{theorem}[Theorem 5.20, Normal basis theorem]
$L$ has a basis of the form $\{\sigma w : \sigma \in G\}$ for some fixed $w$.
\end{theorem}

\begin{theorem}[Theorem 5.21]
For any Galois extension $L/K$, if $G = \mathrm{Gal}(L/K)$ we have $H^1(G, L) = 0$.
\end{theorem}

\begin{proof}
View $L$ as a module over $K[G]$. Then $L \cong K[G]$ so it is a free module (by the normal basis theorem).
\end{proof}

\begin{corollary}[Corollary 5.3]
The normal basis theorem implies $H^i(G, L) = 0$ for all $i > 0$, since this vanishes for any free $K[G]$-module.
\end{corollary}

\section{Galois cohomology}

\begin{definition}[Definition 5.20, Crossed homomorphism]
A homomorphism $G \to M$ written $\sigma \mapsto a_\sigma$ is a crossed homomorphism iff $a_{\sigma\tau} = a_\sigma + \sigma a_\tau$. It is a principal crossed homomorphism iff there is some $b$ such that $a_\sigma = b/\sigma b$.
\end{definition}

\begin{definition}[Definition 5.21, First cohomology group]
The first cohomology group is
\[ H^1(G, M) = \frac{\text{crossed homomorphisms}}{\text{principal crossed homomorphisms}} \]
\end{definition}

\section{Infinite Galois extensions}

\begin{definition}[Definition 5.22, Galois extension, general]
Let $L/K$ be a field extension. This is a Galois extension iff it is normal, separable, and algebraic.
\end{definition}

\begin{definition}[Definition 5.23, Profinite completion]
Let $G$ be a group, and $G_i$ the normal subgroups of $G$ such that $G/G_i$ is finite. Then the profinite completion of $G$ is $\hat{G} = \varprojlim G/G_i$.
\end{definition}

\begin{definition}[Definition 5.24, Krull topology]
Let $G$ be a group, and let $G_i$ run over the normal subgroups of finite index. Equip each $G/G_i$ with the discrete topology, give $\prod G/G_i$ the product topology, and give $\varprojlim G/G_i$ the subset topology. This is the Krull topology.
\end{definition}

\begin{theorem}[Proposition 5.31]
The map $\eta : G \to \hat{G}$ is a homomorphism, and $\mathrm{im}(\eta)$ is dense in $\hat{G}$ equipped with the Krull topology.
\end{theorem}
