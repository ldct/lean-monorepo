\chapter[Rings]{Rings}

\section{Definitions and examples}

\begin{definition}[Definition 2.1, Ring]
A ring is a set $R$ along with two binary operations, $+$ and $\times$, such that:
1. $R$ is an abelian group under $+$
2. $\times$ is associative
3. $a(b + c) = ab + ac$, $(a + b)c = ac + bc$
We also have two optional axioms:
1. $\times$ has an identity
2. $ab = ba$ (commutative rings).
\end{definition}

\begin{definition}[Definition 2.2, Unit]
Let $R$ be a ring. An element $a \in R$ is a unit iff it has a multiplicative inverse in $R$. We denote the collection of all invertible elements of $R$ as $R^\times$.
\end{definition}

\begin{theorem}[Proposition 2.1]
Let $R$ be a ring. The group of units, $R^\times$, is a group. If we let $R$ be a commutative ring, $R^\times$ becomes an abelian group.
\end{theorem}

\begin{definition}[Definition 2.3, Group ring]
Let $G$ be a group and $R$ a commutative ring. The group ring $R[G]$ is the free abelian group with basis $G$, where $\times$ is the group operation on $G$ extended linearly.
\end{definition}

\begin{theorem}[Proposition 2.2]
More generally, for a ring $R$, suppose $e \in R$ is idempotent. Then we have:
\[ R = eR \oplus (1 - e)R \]
both of which are rings. Conversely, in $A \times B$, $(1, 0)$ is idempotent. So the presence of idempotents is equivalent to the ring splitting as a product.
\end{theorem}

\begin{definition}[Definition 2.4, Convolution]
We can think of elements of $R[G]$ as functions from $G \to R$, where $f(g_i) = r_i$. Then the product of $R[G]$ is given by
\[ fh(g) = \sum_{g_1g_2 = g} f(g_1)h(g_2) \]
which is called the convolution of $f$ and $h$.
\end{definition}

\section{Ideals}

\begin{definition}[Definition 2.5, Ideal]
An ideal $I$ of a ring $R$ is a subset of $R$ such that:
1. $I$ contains $0_R$ and is closed under addition and subtraction ($I$ is a normal subgroup of $R$ with respect to addition.)
2. If $r \in I$, and $t \in R$ then $rt, tr \in I$ (stronger than saying that $I$ is closed under $\times$.)
\end{definition}

\section{Integral domains}

\begin{definition}[Definition 2.6, Integral domain]
A ring $R$ is an integral domain iff $1 \neq 0$, it is commutative, and there are no zero divisors in the ring.
\end{definition}

\begin{definition}[Definition 2.7, Euclidean Domain]
A commutative ring $R$ is a Euclidean domain iff it has a function $|\cdot| : R \to \mathbb{N}$ such that given $a, b$ with $b \neq 0$, we can find $r, q$ such that $a = bq + r$ and $|r| < |b|$.
\end{definition}

\begin{definition}[Definition 2.8, Ideal generated by elements]
The ideal generated by elements $g_1, g_2, \ldots$ is the smallest ideal containing these elements. We write this as $(g_1, g_2, \ldots)$.
\end{definition}

\begin{definition}[Definition 2.9, Principal ideal domain]
A principal ideal domain is a commutative ring where all ideals are generated by one element.
\end{definition}

\begin{theorem}[Theorem 2.1]
All Euclidean domains are PIDs.
\end{theorem}

\begin{proof}
Let $I$ be any ideal. We then choose $a \in I$ with $a \neq 0$, and $|a|$ minimal. Then we claim that $I = (a)$.
Suppose $b \in I$. Then $b = aq + r$ with $|r| < |a|$. So $r = b - aq$ means that $r \in I$, and the minimality of $|a|$ forces $r = 0$. This means $b = aq$ for some $q$, and this holds for any $b \in I$, so $I = (a)$.
\end{proof}

\begin{definition}[Definition 2.10, Divides]
Let $a, b \in R$. We say $a$ divides $b$ and write $a \mid b$ iff there exists some $c \in R$ such that $ac = b$.
\end{definition}

\begin{definition}[Definition 2.11, Irreducible]
An element $a \in R$ is called irreducible in $R$ iff it is not 0, it is not a unit, and $a = bc$ implies either $b$ or $c$ is a unit.
\end{definition}

\begin{definition}[Definition 2.12, Prime element]
An element $a \in R$ is called prime iff $a \mid bc$ implies that $a \mid b$ or $a \mid c$.
\end{definition}

\begin{lemma}[Lemma 2.1]
In a principal ideal domain, irreducible elements are also prime.
\end{lemma}

\begin{proof}
Suppose $p$ is irreducible and $p \mid ab$. We want to show that either $p \mid a$ or $p \mid b$. Suppose that $p \nmid a$. Then the ideal $(p, a) = (c)$ for some $c \in R$ since $R$ is a PID. Now $c \mid p$ but since $p$ is irreducible, $c$ must be either a unit, or $p$ up to a unit. If $c$ is a unit multiple of $p$, then since $c \mid a$, we must have $p \mid a$ which is a contradiction. So $c$ is a unit, meaning $(c) = R$.
It follows that there are elements $x, y \in R$ such that $px + ay = 1$. So $pbx + aby = b$ and since both terms on the LHS are divisible by $p$, we have that $p \mid b$ and $p$ is prime as desired.
\end{proof}

\begin{theorem}[Proposition 2.3]
If $R$ is an integral domain, prime elements are irreducible.
\end{theorem}

\begin{proof}
Consider some prime $p$. If $p = ab$, then $p \mid ab$, so WLOG assume $p \mid a$. Then $a = pc$ for some $c \in R \setminus \{0\}$ which means $pc = a = abc$ and $a(1 - bc) = 0$ so $bc = 1$, and $b$ is a unit, making $p$ irreducible.
\end{proof}

\begin{definition}[Definition 2.13, Unique factorization domain]
A commutative ring $R$ is a unique factorization domain iff every element in $R$ can be expressed uniquely as a product of irreducible elements, up to order and unit multiples.
\end{definition}

\begin{theorem}[Proposition 2.4]
If $R$ is a UFD, then every irreducible element is prime.
\end{theorem}

\begin{proof}
Assume $q \in R$ is irreducible. Then if $q \mid ab$ for $a, b \in R$, we have that for some $m \in R$, $mab = q$. Since this is a UFD, we have some unique representation of $m, a, b$ up to units. Therefore, up to units, for irreducible elements $m_1, \ldots, m_i, a_1, \ldots, a_j, b_1, \ldots, b_k \in R$, we have
\[ q = m_1 \cdots m_i a_1 \cdots a_j b_1 \cdots b_k \]
but taking any split of this product, one factor must be a unit since $q$ is irreducible. This means we can split appropriately to find that either $q \mid a$ or $q \mid b$ as desired.
\end{proof}

\begin{theorem}[Theorem 2.2]
Every principal ideal domain is a unique factorization domain.
\end{theorem}

\begin{proof}
Let $R$ be a PID. We first show existence of a factorization of any element into irreducibles. Given any element $a \in R$, if $a$ is not a unit, we can find some irreducible $p \in R$ such that $p \mid a$. Let $a = bc$ for some $b, c \in R$. If $b$ is irreducible we stop, if not we write $b = de$. Repeat until it stops. Will this go on forever?
No. Suppose we have an ideal generated by $a_1, a_2, \ldots$ where $a_1 = a_2'a_2, a_2 = a_3'a_3, \ldots$ with $a_1', a_2', \ldots$ not units. We know there is some $x$ such that $(a_1, a_2, \ldots) = (x)$ since this is a PID. Therefore $x \in (a, a_1, \ldots, a_n)$ for some finite enumeration, so the sequence must stop after finitely many steps.

We now show uniqueness. The key step is that irreducibles are prime. Suppose we have two representations:
\[ a = p_1 \cdots p_m = q_1 \cdots q_n \]
with all $p_i, q_i$ irreducible. Since $p_1$ is prime, $p_1$ divides some $q_i$. But $q_i$ is irreducible, so $q_i$ must be a unit multiple of $p_1$. We can cancel from both sides. Repeating this gives the desired result.
\end{proof}

\begin{theorem}[Theorem 2.3, Fermat's two-square theorem]
Any prime $p \in \mathbb{Z}$ where $p > 0$ and $p \equiv 1 \pmod{4}$ can be uniquely expressed as $a^2 + b^2$ up to sign changes for $a, b$.
\end{theorem}

\begin{proof}
We know $(\mathbb{Z}/p\mathbb{Z})^\times$ is cyclic of order $p - 1 = 4n$ for some $n \in \mathbb{Z}$. We also know it has $-1 \in (\mathbb{Z}/p\mathbb{Z})^\times$ which has order 2. If $g \in (\mathbb{Z}/p\mathbb{Z})^\times$ is the generator, then $g^{4n} = 1$ which means $-1 \equiv g^{2n} \pmod{p}$, so $-1$ is a square modulo $p$. This means that for some $n, a \in \mathbb{Z}$, $-1 = a^2 - np$. Therefore $np = a^2 + 1 = (a + i)(a - i)$ in $\mathbb{Z}[i]$. In particular, $p \mid (a + i)(a - i)$ but does not divide either of $(a \pm i)$. Therefore $p$ is not prime and therefore not irreducible in $\mathbb{Z}[i]$. This means for some $a, b \in \mathbb{Z}$ we have $p = (a + bi)(a - bi)$. Therefore $p = a^2 + b^2$.
To see uniqueness, assume $p = x^2 + y^2$. Then $p = (x + iy)(x - iy)$ so $x + iy = u(a + bi)$ where $u$ is some unit of $\mathbb{Z}[i]$ since $\mathbb{Z}[i]$ is a UFD. This means $x = \pm a$ and $y = \pm b$ as desired.
\end{proof}

\section{Ideals of commutative rings}

\begin{definition}[Definition 2.14, Integral domain]
A commutative ring $R$ is an integral domain iff for $a, b \in R$, $ab = 0 \implies a = 0$ or $b = 0$.
\end{definition}

\begin{definition}[Definition 2.15, Field]
A commutative ring is a field iff it has no zero divisors, and every element has a multiplicative inverse.
\end{definition}

\begin{definition}[Definition 2.16, Maximal Ideal]
An ideal $I$ of a ring $R$ is a maximal ideal iff it is the maximal element of the proper ideals of $R$.
\end{definition}

\begin{definition}[Definition 2.17, Prime ideal]
An ideal $I$ of a ring $R$ is a prime ideal iff $ab \in I \implies a \in I$ or $b \in I$.
\end{definition}

\begin{theorem}[Theorem 2.4]
An ideal $I$ of a ring $R$ is maximal iff $R/I$ is a field.
\end{theorem}

\begin{theorem}[Theorem 2.5]
An ideal $I$ of a ring $R$ is prime iff $R/I$ is an integral domain.
\end{theorem}

\begin{theorem}[Theorem 2.6]
Maximal ideals are always prime ideals (but not the other way around).
\end{theorem}

\begin{theorem}[Theorem 2.7]
The maximal ideal of a field is always 0.
\end{theorem}

\begin{proof}
Assume $F$ is a field. This means 0 is the maximal ideal. Now assume 0 is the maximal ideal of some ring $R$. Then $R = (a)$ for some nonzero $a$. So $ab = 1$ for some $b \in R$ which means $a$ has an inverse.
\end{proof}

\begin{theorem}[Proposition 2.5]
The following inclusions hold:
\[ \text{Crings} \supset \text{ID} \supset \text{UFD} \supset \text{PID} \supset \text{ED} \supset \text{Fields} \]
\end{theorem}

\begin{definition}[Definition 2.18, Maximal element]
The maximal element of some ordered set is an element $a \in S$ such that there does not exist an element such that $b > a$.
\end{definition}

\begin{definition}[Definition 2.19, Partial Order]
Given a set $S$, a partial order is a binary relation $\leq : S \times S \to S$ such that for all $x, y, z \in S$, we have:
1. Reflexive: $x \leq x$
2. Transitive: $x \leq y$ and $y \leq z$ implies $x \leq z$
3. Antisymmetric: $a \leq b$ and $b \leq a$ implies $a = b$
\end{definition}

\begin{lemma}[Lemma 2.2, Zorn's Lemma]
Consider a nonempty partially ordered set $S$. If it is the case that every totally ordered subset of $S$ has an upper bound, then we have that $S$ itself has a maximal element.
\end{lemma}

\begin{proof}
First, we take $s_0 \in S$ since $S$ is nonempty. Then $\{s_0\}$ is totally ordered, so we have an upper bound $s_1$. If $s_0$ is not maximal, then $s_1 > s_0$. Repeat with $\{s_0, s_1\}$. Continue infinitely many times. At this point we have some upper bound $s_\omega$ of $\{s_0, s_1, \ldots\}$. We find some $s_\alpha$ for every ordinal $\alpha$ but the set of ordinals is a proper class, and since $S$ is a set, it is smaller than the class of ordinals. Therefore $S$ is exhausted before the ordinals are and we get a maximal element.
\end{proof}

\begin{lemma}[Lemma 2.3]
The union of a totally ordered set of ideals of some ring $R$ is also an ideal.
\end{lemma}

\begin{proof}
Take some totally ordered set of ideals $\{I_\alpha\}$ of a ring $R$. If $a, b \in \cup I_\alpha$ then for some $\alpha_1, \alpha_2$, $a \in I_{\alpha_1}$ and $b \in I_{\alpha_2}$. Since the set is totally ordered by inclusion, WLOG $I_{\alpha_1} \subseteq I_{\alpha_2}$ so $ab \in I_{\alpha_2}$.
\end{proof}

\begin{theorem}[Theorem 2.8]
If $I$ is a proper ideal of some ring $R$, then $I$ is contained in some maximal ideal, so there exists a maximal ideal, unless $R = 0$.
\end{theorem}

\begin{proof}
Assuming Zorn's lemma, the set of all proper ideals of any ring $R$ containing some ideal $I$ satisfies the hypotheses of Zorn's lemma. $S$ is partially ordered by inclusion. It is nonempty because $I \in S$. Every totally ordered subset has an upper bound (the union, by Lemma 2.3). Therefore $S$ has a maximal element.
\end{proof}

\begin{corollary}[Corollary 2.1]
The intersection of all prime ideals of a ring is the set of all nilpotent elements of the ring.
\end{corollary}

\begin{proof}
If $p$ is a prime ideal of some ring $R$, then $x^n \in p$ means either $x^{n-1}$ or $x$ is in $p$. Repeating, if $x$ is nilpotent, it is in every prime ideal.
Now suppose $x$ is not nilpotent. Define $M := \{1, x, x^2, \ldots\} \not\ni 0$ because $x$ is not nilpotent. Let $S$ be the set of ideals disjoint from $M$. $S$ is partially ordered by inclusion and nonempty since $(0) \in S$. Every totally ordered subset has an upper bound (the union). By Zorn's lemma, $S$ has a maximal element $I$.
We show $I$ is prime. Suppose $a, b \notin I$. Then $(I, a) \supsetneq I$ so it contains an element of $M$: $i_1 + sa = x^n$. Similarly $(I, b)$ contains $i_2 + tb = x^m$. Then $i_1i_2 + i_1tb + i_2sa + stab = x^{m+n} \in M$, so $ab \notin I$ because otherwise the entire expression would be in $I$, contradicting that it is in $M$. Thus $I$ is prime.
\end{proof}

\section{Localization}

\begin{definition}[Definition 2.20, Quotient ring / Localization]
Given a commutative ring $R$, and a multiplicative subset $S \subseteq R$, the ring consisting of all elements of $S$ and their multiplicative inverses is called the quotient ring (or localization) and is written $R[S^{-1}]$.
\end{definition}

\begin{theorem}[Proposition 2.6]
Let $R$ be a commutative ring, $S \subseteq R$. Also let $1 \in S$, and $S$ is multiplicatively closed with no zero divisors. Then the relation $(r_1, s_1) \sim (r_2, s_2) \iff r_1s_2 = r_2s_1$ is an equivalence relation, and the equivalence classes under this relation form a quotient ring $R[S^{-1}]$.
\end{theorem}

\begin{proof}
Take all pairs $(r, s)$ such that $r \in R$, $s \in S$. We define the relation as in the statement. For transitivity: suppose $(r_1, s_1) \sim (r_2, s_2)$ and $(r_2, s_2) \sim (r_3, s_3)$. Then $r_1s_2s_3 = r_2s_1s_3 = r_3s_1s_2$, so $0 = s_2(r_1s_3 - r_3s_1)$. Since there are no zero divisors, $r_1s_3 = r_3s_1$ giving $(r_1, s_1) \sim (r_3, s_3)$. The map $R \to R[S^{-1}]$ is injective since if $r \mapsto (r, 1)$ and $(r, 1) \sim (0, 1)$ then $r = 0$.
\end{proof}

\begin{theorem}[Proposition 2.7]
Let $R$ be a commutative ring, $S \subseteq R$ be a multiplicative subset and $1 \in S$. Then if we take $(r_1, s_1) \sim (r_2, s_2)$ iff there exists some $s_3 \in S$ such that $s_3(r_2s_1 - s_2r_1) = 0$, this gives us a quotient ring $R[S^{-1}]$ with the following properties:
1. There exists a homomorphism from $R$ to $R[S^{-1}]$.
2. The image of all elements of $S$ in $R[S^{-1}]$ are invertible in $R[S^{-1}]$.
3. $R[S^{-1}]$ is the universal ring with these properties.
\end{theorem}

\begin{proof}
This proposition does not assume the absence of zero divisors. We adapt the equivalence relation from the previous proposition. Let $I := \{x \in R \mid \exists s \in S, xs = 0\}$. This is closed under addition. Consider $R/I$, and define $\bar{S}$ to be the image of $S$ in $R/I$. Now form $(R/I)[\bar{S}^{-1}]$ as in the non-zero divisor case. The kernel of the map from $R$ to $R[S^{-1}]$ is $I$, the set of elements killed by something in $S$.
\end{proof}
